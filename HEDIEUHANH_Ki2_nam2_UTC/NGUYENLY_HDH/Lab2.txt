	esc + ":q!": thoat va quay lai terminal(chua luu file)
	esc escape + ":wq!": thoat va luu file

	esc + "i": insert tai vi tri con tro
	esc + "a": insert sau vi tri con tro

	esc + "d(so tu muon xoa)w": xoa tu tai vi tri con tro(dua con tro ve dau tu 	can xoa)
	vd d1w, d2w, d3w

	esc + "dd": xoa mot dong

	esc + "yy": copy dong hien tai

	esc + "p": paste len vi tri con tro

	esc + "/" + tu_muon_tim_kiem: 
	esc + ":set nu" : hien thi so dong

	esc ":8" : chuyen den dong thu 8
